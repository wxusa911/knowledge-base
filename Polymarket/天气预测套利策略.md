def get_weather_markets():

        "https://gamma-api.polymarket.com/markets",

            "active": "true",
            "limit": 50
        }
    )
    return response.json()

# 2. 分析市场定价
def analyze_market(market):
    # 找出定价偏离度高的市场
    # 例如：气象预报说80%下雨，但市场只定价40%
    
    weather_data = get_weather_forecast(market['question'])
    market_price = float(market.get('bestAsk', 0.5))
    
    if abs(weather_data.probability - market_price) > 0.2:
        return {
            "opportunity": True,
            "edge": weather_data.probability - market_price,
            "direction": "YES" if weather_data.probability > market_price else "NO"
        }
    return {"opportunity": False}

# 3. 下单执行
def place_bet(client, token_id, direction, size=1):
    order = OrderArgs(
        token_id=token_id,
        price=0.5,  # 挂单价格
        size=size,
        side=BUY if direction == "YES" else SELL,
        order_type=OrderType.GTC
    )
    return client.create_order(order)
